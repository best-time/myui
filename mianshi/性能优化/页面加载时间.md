|指标	|含义	|达标阈值|
|---|---|---|
|FP |首次绘制	页面第一次出现像素，白屏结束	|<1.8s|
|FCP |首次内容绘制	首次渲染文字 / 图片，用户看到有效内容	|<1.8s|
|LCP |最大内容绘制	页面主内容加载完成，核心首屏指标	|<2.5s|
|FID / INP |交互延迟	用户点击、输入时页面响应速度	|<100ms|
|CLS | 布局偏移	页面加载中元素抖动程度	|<0.1|

### 原生Performance Api

1. 页面基础加载耗时
```javascript
function getPageTiming() {
  const t = performance.timing;
  const timing = {
    // 白屏时间 = 首次渲染 - 页面开始
    whiteScreen: t.domLoading - t.navigationStart,
    // DOM解析完成时间
    domReady: t.domContentLoadedEventEnd - t.navigationStart,
    // 页面完全加载（含所有图片资源）
    pageLoad: t.loadEventEnd - t.navigationStart,
    // 网络分层耗时
    dns: t.domainLookupEnd - t.domainLookupStart,
    tcp: t.connectEnd - t.connectStart,
    request: t.responseEnd - t.requestStart,
    redirect: t.redirectEnd - t.redirectStart,
  }
  return timing;
}
```
2. web Vitals
```javascript
// 统一收集指标
const perfData = {};

// LCP 最大内容绘制
new PerformanceObserver((list) => {
  const entry = list.getEntries()[list.getEntries().length - 1];
  perfData.LCP = entry.startTime;
}).observe({ type: 'largest-contentful-paint', buffered: true });

// INP 交互延迟（替代废弃FID）
new PerformanceObserver((list) => {
  const entries = list.getEntries();
  const lastEntry = entries[entries.length - 1];
  perfData.INP = lastEntry.processingDuration;
}).observe({ type: 'interaction-next-paint', buffered: true });

// CLS 布局偏移
let cls = 0;
new PerformanceObserver((list) => {
  for (const entry of list.getEntries()) {
    if (!entry.hadRecentInput) cls += entry.value;
  }
  perfData.CLS = cls;
}).observe({ type: 'layout-shift', buffered: true });

// 页面完全加载后统一上报
window.addEventListener('load', () => {
  setTimeout(() => {
    const timing = getPageTiming();
    const report = {
      url: location.href,
      userAgent: navigator.userAgent,
      timestamp: Date.now(),
      ...timing,
      ...perfData
    };
    // 上报接口
    fetch('/api/perf/report', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(report),
      keepalive: true // 页面关闭也能发送完成
    })
  }, 3000);
})
```
